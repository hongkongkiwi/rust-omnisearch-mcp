13#[async_trait]
14pub trait CacheProvider: Send + Sync {
15    async fn get(&self, key: &str) -> Result<Option<CacheValue>>;
16    async fn set(&self, key: &str, value: CacheValue, ttl: Duration) -> Result<()>;
17    async fn delete(&self, key: &str) -> Result<()>;
18    async fn clear(&self) -> Result<()>;
19    async fn size(&self) -> Result<usize>;
20}

22pub struct MemoryCache {
23    cache: MokaCache<String, CacheValue>,
24}
25
26impl MemoryCache {
27    pub fn new(config: &CacheConfig) -> Self {
28        let cache = MokaCache::builder()
29            .max_capacity(config.max_entries as u64)
30            .time_to_live(Duration::from_secs(config.ttl_seconds))
31            .build();
32
33        Self { cache }
34    }
35}
36
37#[async_trait]
38impl CacheProvider for MemoryCache {
39    async fn get(&self, key: &str) -> Result<Option<CacheValue>> {
40        let value = self.cache.get(key).await;
41        debug!(
42            "Cache {} for key: {}",
43            if value.is_some() { "hit" } else { "miss" },
44            key
45        );
46        Ok(value)
47    }
48
49    async fn set(&self, key: &str, value: CacheValue, _ttl: Duration) -> Result<()> {
50        let len = value.len();
51        self.cache.insert(key.to_string(), value).await;
52        debug!("Cached {} results for key: {}", len, key);
53        Ok(())
54    }
55
56    async fn delete(&self, key: &str) -> Result<()> {
57        self.cache.remove(key).await;
58        debug!("Removed cache entry for key: {}", key);
59        Ok(())
60    }
61
62    async fn clear(&self) -> Result<()> {
63        self.cache.invalidate_all();
64        info!("Cleared memory cache");
65        Ok(())
66    }
67
68    async fn size(&self) -> Result<usize> {
69        Ok(self.cache.entry_count() as usize)
70    }
71}

73pub struct CacheManager {
74    provider: Box<dyn CacheProvider>,
75    enabled: bool,
76}
77
78impl CacheManager {
79    pub async fn new() -> Result<Self> {
80        let config = &CONFIG.cache;
81
82        if !config.enabled {
83            info!("Cache is disabled");
84            let dummy_cache = MemoryCache::new(config);
85            return Ok(Self {
86                provider: Box::new(dummy_cache),
87                enabled: false,
88            });
89        }
90
91        let provider: Box<dyn CacheProvider> = match config.cache_type {
92            CacheType::Memory => {
93                info!("Using memory cache with {} max entries", config.max_entries);
94                Box::new(MemoryCache::new(config))
95            }
96        };
97
98        Ok(Self {
99            provider,
100            enabled: config.enabled,
101        })
102    }
103
104    pub async fn get(&self, key: &str) -> Result<Option<CacheValue>> {
105        if !self.enabled {
106            return Ok(None);
107        }
108        self.provider.get(key).await
109    }
110
111    pub async fn set(&self, key: &str, value: CacheValue, ttl: Duration) -> Result<()> {
112        if !self.enabled {
113            return Ok(());
114        }
115        self.provider.set(key, value, ttl).await
116    }
117
118    pub async fn delete(&self, key: &str) -> Result<()> {
119        if !self.enabled {
120            return Ok(());
121        }
122        self.provider.delete(key).await
123    }
124
125    pub async fn clear(&self) -> Result<()> {
126        if !self.enabled {
127            return Ok(());
128        }
129        self.provider.clear().await
130    }
131
132    pub async fn size(&self) -> Result<usize> {
133        if !self.enabled {
134            return Ok(0);
135        }
136        self.provider.size().await
137    }
138
139    pub fn generate_cache_key(provider: &str, query: &str, limit: Option<usize>) -> String {
140        format!("{}:{}:{}", provider, query, limit.unwrap_or(10))
141    }
142}

168    #[tokio::test]
169    async fn test_memory_cache_operations() {
170        let config = CacheConfig {
171            enabled: true,
172            cache_type: CacheType::Memory,
173            ttl_seconds: 60,
174            max_entries: 100,
175        };
176
177        let cache = MemoryCache::new(&config);
178        let test_results = create_test_results();
179
180        // Test set and get
181        cache
182            .set("test_key", test_results.clone(), Duration::from_secs(60))
183            .await
184            .unwrap();
185
186        let retrieved = cache.get("test_key").await.unwrap();
187        assert!(retrieved.is_some());
188        assert_eq!(retrieved.unwrap().len(), 2);
189
190        // Test size (moka cache may not immediately report size)
191        let _size = cache.size().await.unwrap();
192        // Cache should exist and report size (size is always >= 0 for usize)
193
194        // Test delete
195        cache.delete("test_key").await.unwrap();
196        let retrieved = cache.get("test_key").await.unwrap();
197        assert!(retrieved.is_none());
198
199        // Test clear
200        cache
201            .set("key1", test_results.clone(), Duration::from_secs(60))
202            .await
203            .unwrap();
204        cache
205            .set("key2", test_results.clone(), Duration::from_secs(60))
206            .await
207            .unwrap();
208
209        cache.clear().await.unwrap();
210        let retrieved1 = cache.get("key1").await.unwrap();
211        let retrieved2 = cache.get("key2").await.unwrap();
212        assert!(retrieved1.is_none());
213        assert!(retrieved2.is_none());
214    }
