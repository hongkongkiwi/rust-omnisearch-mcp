15#[derive(Clone)]
16pub struct RateLimiterManager {
17    limiters: Arc<RwLock<HashMap<String, Arc<ProviderRateLimiter>>>>,
18    enabled: bool,
19}

21impl Default for RateLimiterManager {
22    fn default() -> Self {
23        Self::new()
24    }
25}

27impl RateLimiterManager {
28    pub fn new() -> Self {
29        let enabled = CONFIG.rate_limiting.enabled;
30
31        Self {
32            limiters: Arc::new(RwLock::new(HashMap::new())),
33            enabled,
34        }
35    }
36
37    pub async fn get_or_create_limiter(&self, provider: &str) -> Result<Arc<ProviderRateLimiter>> {
38        if !self.enabled {
39            // Return a very permissive rate limiter when disabled
40            let quota = Quota::per_minute(NonZeroU32::new(u32::MAX).unwrap());
41            return Ok(Arc::new(GovernorLimiter::direct(quota)));
42        }
43
44        let mut limiters = self.limiters.write().await;
45
46        if let Some(limiter) = limiters.get(provider) {
47            return Ok(Arc::clone(limiter));
48        }
49
50        // Get provider-specific rate limit from config
51        let rate_limit = self.get_provider_rate_limit(provider);
52        let quota = Quota::per_minute(
53            NonZeroU32::new(rate_limit)
54                .ok_or_else(|| eyre!("Rate limit must be greater than 0"))?,
55        );
56
57        let limiter = Arc::new(GovernorLimiter::direct(quota));
58        limiters.insert(provider.to_string(), Arc::clone(&limiter));
59
60        debug!(
61            "Created rate limiter for provider '{}' with {} requests/minute",
62            provider, rate_limit
63        );
64
65        Ok(limiter)
66    }
67
68    fn get_provider_rate_limit(&self, provider: &str) -> u32 {
69        match provider {
70            "tavily" => CONFIG.providers.tavily.rate_limit,
71            "google" => CONFIG.providers.google.rate_limit,
72            "reddit" => CONFIG.providers.reddit.rate_limit,
73            "duckduckgo" => CONFIG.providers.duckduckgo.rate_limit,
74            "baidu" => CONFIG.providers.baidu.rate_limit,
75            "exa" => CONFIG.providers.exa.rate_limit,
76            "brave" => CONFIG.providers.brave.rate_limit,
77            "kagi" => CONFIG.providers.kagi.rate_limit,
78            "perplexity" => CONFIG.providers.perplexity.rate_limit,
79            "jina" => CONFIG.providers.jina.rate_limit,
80            "firecrawl" => CONFIG.providers.firecrawl.rate_limit,
81            "brightdata" => CONFIG.providers.brightdata.rate_limit,
82            _ => {
83                warn!("Unknown provider '{}', using default rate limit", provider);
84                60 // Default rate limit
85            }
86        }
87    }
88
89    pub async fn check_rate_limit(&self, provider: &str) -> Result<()> {
90        if !self.enabled {
91            return Ok(());
92        }
93
94        let limiter = self.get_or_create_limiter(provider).await?;
95
96        match limiter.check() {
97            Ok(_) => {
98                debug!("Rate limit check passed for provider: {}", provider);
99                Ok(())
100            }
101            Err(_) => {
102                warn!("Rate limit exceeded for provider: {}", provider);
103                Err(eyre!("Rate limit exceeded for provider: {}", provider))
104            }
105        }
106    }
107
108    pub async fn wait_for_rate_limit(&self, provider: &str) -> Result<()> {
109        if !self.enabled {
110            return Ok(());
111        }
112
113        let limiter = self.get_or_create_limiter(provider).await?;
114
115        limiter.until_ready().await;
116        debug!("Rate limit wait completed for provider: {}", provider);
117        Ok(())
118    }
119
120    pub async fn reset_limiter(&self, provider: &str) -> Result<()> {
121        let mut limiters = self.limiters.write().await;
122        limiters.remove(provider);
123        debug!("Reset rate limiter for provider: {}", provider);
124        Ok(())
125    }
126
127    pub async fn get_limiter_stats(&self, provider: &str) -> Result<Option<RateLimiterStats>> {
128        if !self.enabled {
129            return Ok(None);
130        }
131
132        let limiters = self.limiters.read().await;
133        if let Some(_limiter) = limiters.get(provider) {
134            // Governor rate limiter doesn't provide snapshot functionality in this version
135            // Return basic stats
136            Ok(Some(RateLimiterStats {
137                provider: provider.to_string(),
138                remaining_burst: 0, // Not available in this version
139                next_replenishment: Some(Duration::from_secs(0)), // Not available
140            }))
141        } else {
142            Ok(None)
143        }
144    }
145}

154// Global rate limiter manager

157pub static RATE_LIMITER_MANAGER: Lazy<RateLimiterManager> = Lazy::new(RateLimiterManager::new);

159// Convenience functions
160pub async fn check_rate_limit(provider: &str) -> Result<()> {
161    RATE_LIMITER_MANAGER.check_rate_limit(provider).await

164pub async fn wait_for_rate_limit(provider: &str) -> Result<()> {

166}

170}
